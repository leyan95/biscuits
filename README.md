Add it in your root build.gradle at the end of repositories:

```groovy
  repositories {

    maven { url 'https://jitpack.io' }
  }
}

dependencies {
        implementation 'com.github.leyan95:pocket:0.5.6-PRE'
        implementation 'com.github.leyan95:skeleton:0.6.0-PRE'
}  
```
